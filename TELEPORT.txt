gui.Parent = player:WaitForChild("PlayerGui")

local button = Instance.new("TextButton")

button.TextSize = 20

-- FunÃ§Ã£o de ativaÃ§Ã£o
button.Activated:Connect(function()
	-- Equipa a RocketLauncher
	local backpack = player:FindFirstChild("Backpack")
	local tool = backpack and backpack:FindFirstChild("RocketLauncher")

	if tool then
		tool.Parent = player.Character
	else
		warn("RocketLauncher nÃ£o encontrada na mochila!")
		return
	end

	-- Teleporta para um Spawn aleatÃ³rio
	local spawnPoints = {}
	for _, obj in ipairs(workspace:GetChildren()) do
		if obj:IsA("SpawnLocation") then
			table.insert(spawnPoints, obj)
		end
	end

	if #spawnPoints > 0 then
		local spawn = spawnPoints[math.random(1, #spawnPoints)]
		local hrp = player.Character and player.Character:FindFirstChild("HumanoidRootPart")
		if hrp then
			hrp.CFrame = spawn.CFrame + Vector3.new(0, 5, 0)
		end
	end

	-- Simula clique automaticamente
	local mouse = player:GetMouse()
	while tool.Parent == player.Character do
		task.wait(0.5) -- Ajusta a frequÃªncia de tiro
		mouse.Button1Down:Fire()
	end
