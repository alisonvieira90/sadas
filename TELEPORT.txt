local Players = game:GetService("Players")
local player = Players.LocalPlayer

-- Esperar o personagem carregar
local function getHRP()
	local char = player.Character or player.CharacterAdded:Wait()
	return char:WaitForChild("HumanoidRootPart")
end

local hrp = getHRP()

local gui = Instance.new("ScreenGui")
gui.Parent = player:WaitForChild("PlayerGui")

local button = Instance.new("TextButton")
button.Size = UDim2.new(0, 200, 0, 50)
button.Position = UDim2.new(0.5, -100, 0.8, 0)
button.Text = "Teleportar pra TPS"
button.Parent = gui

-- Clique = teleportar
button.Activated:Connect(function()
	local folder = workspace:FindFirstChild("Balls")
	if folder then
		local TPS = folder:FindFirstChild("TPS")
		if TPS then
			hrp.CFrame = TPS.CFrame + Vector3.new(0, 5, 0)
		else
			warn("TPS não encontrado dentro de Balls.")
		end
	else
		warn("Folder 'Balls' não encontrado no Workspace.")
	end
end)
